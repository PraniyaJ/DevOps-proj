//jenkins pipeline

pipeline {
    agent any
    
    stages {
        stage('SCM Checkout') {
            steps {
                retry(3) {
                    git branch: 'main', url: 'https://github.com/PraniyaJ/devopsproj'
                }
            }
        }
        
        stage('Build Frontend Docker Image') {
            steps {
                dir('frontend') {
                    sh "docker build -t praniya/devopsproj-frontend:${BUILD_NUMBER} ."
                }
            }
        }
        
        stage('Build Backend Docker Image') {
            steps {
                dir('backend') {
                    sh "docker build -t praniya/devopsproj-backend:${BUILD_NUMBER} ."
                }
            }
        }
        
        stage('Login to Docker Hub') {
            steps {
                withCredentials([string(credentialsId: 'devops-projpassword', variable: 'DOCKER_PASSWORD')]) {
                    script {  
                        sh "docker login -u praniya -p '${DOCKER_PASSWORD}'"
                    }
                }
            }
        }
        
        stage('Push Frontend Image') {
            steps {
                sh "docker push praniya/devopsproj-frontend:${BUILD_NUMBER}"
            }
        }
        
        stage('Push Backend Image') {
            steps {
                sh "docker push praniya/devopsproj-backend:${BUILD_NUMBER}"
            }
        }
    }
    
    post {
        always {
            sh 'docker logout'
        }
    }
}
